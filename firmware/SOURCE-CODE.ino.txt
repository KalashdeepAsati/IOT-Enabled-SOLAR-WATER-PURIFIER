ser.println("AT");

delay(100);

delay(100);

void loop()

{
volt = analogRead(A4);
volt = volt*0.0594;
charge = analogRead(A5);
charge = charge*0.0594;

gravityTds.setTemperature(temperature); // Set the temperature for compensation
gravityTds.update();                  // Sample and calculate
tdsValue = gravityTds.getTdsValue();  // Get the TDS value in ppm
tds=tdsValue;
Serial.print("TDS Value:");
Serial.print(tdsValue, 0);
Serial.println("ppm");




tru();
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("TDS:"); 
lcd.print(tds);
lcd.print("ppm");
lcd.setCursor(0, 1);
lcd.print("TURBIDITY:");
lcd.print(gas);
lcd.print(" NTU");
delay(1500);
lcd.clear();
lcd.setCursor(0,0);
lcd.print("Battery=");
lcd.print(volt);
lcd.print(" V");

lcd.setCursor(0,1);
lcd.print("Solar=");
lcd.print(charge);
lcd.print(" V");
//delay(1000);

if (charge>=11)
{
lcd.clear();
lcd.setCursor(0,0);
lcd.print("   BATTERY ");
lcd.setCursor(0,1);
lcd.print(" CHARGING ON"); 
delay(2000);
}

state1=volt;
state2=charge;
state3=tds;
state4=gas;
esp_8266();
} 



void esp_8266()

{

   // TCP connection AT+CIPSTART=4,"TCP","184.106.153.149",80 

    String cmd = "AT+CIPSTART=4,\"TCP\",\"";

    cmd += "184.106.153.149"; // api.thingspeak.com

    cmd += "\",80";

    ser.println(cmd);

    Serial.println(cmd); 

    if(ser.find("Error"))

    {

      Serial.println("AT+CIPSTART error");
      return;

    }                                   // prepare GET string GET https://api.thingspeak.com/update?api_key=LHAG4NSIYJ5UWS6U&field1=0\r\n\r\n

 String getStr = "GET /update?api_key=";
 getStr += apiKey;

getStr +="&field1=";
getStr +=String(state1);

getStr +="&field2=";
getStr +=String(state2);

getStr +="&field3=";
getStr +=String(state3);

getStr +="&field4=";
getStr +=String(state4);


  getStr += "\r\n\r\n";    // send data length

  cmd = "AT+CIPSEND=4,";

  cmd += String(getStr.length());

  ser.println(cmd);

  Serial.println(cmd);

  delay(1000);

ser.print(getStr);

Serial.println(getStr);   // thingspeak needs 15 sec delay between updates
delay(6000);  
  
lcd.clear();
lcd.print("DATA UPLOADING");
lcd.setCursor(0,1);
lcd.print("ON SERVER.....");
delay(2000);
}





void tru() 
{

gas = analogRead(A1);
gas= gas*0.1;
gas=102-gas;
gas=gas-7;

if (gas<20)
{
lcd.clear();
lcd.print("LOW TURBIDITY");
lcd.setCursor(0,1);
lcd.print("CAN BE DRINK");
delay(2000);
}       

else if ((gas<=40) && (gas>=20))
{
lcd.clear();
lcd.print("MEDIUM TURBIDITY");
lcd.setCursor(0,1);
lcd.print("USE FOR WASHING");
delay(2000);
} 

else if (gas>=40)
{
digitalWrite(buzz,1);
lcd.clear();
lcd.print("HIGH TURBIDITY");
lcd.setCursor(0,1);
lcd.print("POOR QUALITY");
delay(2000);
digitalWrite(buzz,0);
} 

}
